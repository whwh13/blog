    思路：

    获取一个输入，当输入为 # 时结束循环。当输入为 \n 时直接获取下一个输入

    当输入为 ‘b’，结束循环

    当输入为 ‘h’，执行 Step 3

    均不是时，执行 Step 2, Step 3

    #include<stdio.h>
int main(void)
{
    char ch;

    while ((ch = getchar()) != '#')
    {
        switch(ch)
        {
            case '\n': break;
            case 'c':
            case 'b':
                printf("Step 1\n");
                break;
            case 'h':
                printf("Step 1\nStep 3\n");
                break;
            default:
                printf("Step 1\nStep 2\nStep 3\n");
                break;
        }
        if (ch = 'b')
            break;
    }
}

    #include<stdio.h>
int main(void)
{
    char ch;

    while ((ch = getchar()) != '#')
    {
        if ( ch != '\n');
        {
            printf("Step1\n");
            if (ch = 'b')
                break;
            else if (ch != 'c')
            {
                if (ch != 'h')
                    printf("Step 2\n");
                printf("Step 3\n");
            }
        }
    }
    printf("Done\n");
}
